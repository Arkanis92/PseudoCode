1. A line of people at an amusement park ride.

class parkRideLine {
  function newPerson(arr, personName) {
    return arr.push(personName);
  }

  function findPerson(arr, personName) {
    for ( let i = 0 ; i <= arr.length ; i++ ) {
      if ( personName === arr[i] ) {
        return i + 1;
      }
    }
    return null;
  }

  function cutLine(array, target, personName) {
    let linePosition = findPerson(array, target);
    if ( linePosition !== null ){
      array.splice( linePosition, 0, personName );
      return array;
    } else {
      return null;
    }
  }

  function leaveLine(arr, personName) {
    let linePosition = findPerson(arr, personName);
    if ( linePosition !== null ){
      arr.splice(linePosition, 1);
      return arr;
    } else {
      return null;
    }
  }
}

2. Provide another real-world example that you can model using a data structure.


Prisoners on one floor of a jail block could be stored in an array based on which cells they're in.

3. How would your data structure allow developers to access and manipulate the data?

The prisoners are stored in an array corresponding to their cell number (i.e. John Smith is in cell 1, so he'd be placed in index 0 of the array).
Developers would be able to update who is in what cell based on their position in the array as well as utilize methods to delete, add, and replace prisoners as they come and go.
